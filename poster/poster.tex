% ── 1. Motivation ───────────────────────────────────────
\block{1.~Motivation}{
    \textbf{Variational Autoencoders} learn compact latent representations
    enabling reconstruction, synthesis, and morphing between sounds -
    but high-quality waveform reconstruction remains hard.

    \medskip
    \vspace{4mm}
    \textbf{Key Question:} What is the minimal VAE architecture that yields
    convincing audio and a musically useful latent space?

    \medskip
    \vspace{4mm}
    \textbf{Creative Potential:}
    The latent space as a synthesis interface - morph between sounds
    in a learned latent space.

    \medskip
    \vspace{4mm}
    \textbf{Progression:}
    \begin{itemize}[leftmargin=1em, noitemsep, topsep=1pt]
        \item \textbf{SpecVAE} - spectrogram-domain baseline, 29 files
        \item \textbf{RawAudioVAE (small)} - raw waveform, 29 files
        \item \textbf{RawAudioVAE (ESC-50)} - 2000 clips, 50 classes
        \item \textbf{RawAudioVAE (Medley-Solos-DB)} - 21k clips, 8 tonal instruments \textit{(in progress)}
    \end{itemize}
}

        \vspace{-6mm}
        \includegraphics[width=0.95\linewidth]{rawvae_small/interp_edit.png}
        \vspace{-6mm}

    \textbf{Slerp} (spherical linear interpolation) from $t{=}0$ to $t{=}1$ between two files' latent sequences.
    Spectrograms evolve smoothly, indicating perceptual coherence in the learned latent space.

    \medskip
    \vspace{2mm}
    \textbf{Mean + offset:} decomposes $z_i = \mu_{\text{global}} + \delta_i$;
    slerp the mean, lerp the offsets - separating timbral character from temporal structure.
}

% ── 2. Architecture Overview ────────────────────────────
\block{2.~Architecture Overview}{
    \begin{multicols}{2}

    \textcolor{accentblue}{\textbf{SpecVAE}}
    \begin{itemize}[leftmargin=1em, noitemsep, topsep=1pt]
        \item \textbf{Input:} log-mel spectrogram (64\,bins $\times$ T\,frames)
        \item \textbf{Encoder:} Conv1d stack $\to$ FC $\to$ $\mu, \sigma$
        \item \textbf{Latent:} $z \in \mathbb{R}^{32}$ per 500\,ms single-vector chunk
        \item \textbf{Decoder:} FC $\to$ Conv1d $\to$ spec $\to$ Griffin-Lim
        \item \textbf{Loss:} MSE on mel spectrogram + $\beta$-KL
    \end{itemize}

    \columnbreak

    \textcolor{accentorange}{\textbf{RawAudioVAE}}
    \begin{itemize}[leftmargin=1em, noitemsep, topsep=1pt]
        \item \textbf{Input:} raw waveform ($1 \times 16{,}384$ samples, 22\,kHz)
        \item \textbf{Encoder:} Conv1d ($\times$4, stride 4) + dilated residual
        \item \textbf{Latent:} $z \in \mathbb{R}^{32 \times 64}$ - temporal sequence, $\approx$86\,Hz
        \item \textbf{Decoder:} Conv1d (mirror encoder) $\to$ waveform
        \item \textbf{Loss:} multi-res STFT (2048, 1024, 512, 256) + $\beta$-KL
    \end{itemize}

    \end{multicols}

    \medskip
    RawAudioVAE's \textbf{temporal latent} (64 frames, $\approx$86\,Hz) preserves within-chunk dynamics lost by SpecVAE's single vector. Free bits ($\lambda = 0.25$) prevent posterior collapse.
}

% ── 9. Conclusions & Future Work ────────────────────────
\block{9.~Conclusions \& Future Work}{
    \textcolor{accentblue}{\textbf{What worked:}}
    \begin{itemize}[leftmargin=1em, noitemsep, topsep=1pt]
        \item RawAudioVAE with multi-resolution STFT loss trains stably and
        improves over SpecVAE baseline.
        \item Larger dataset (ESC-50) generalizes better.
        %\item Free bits + $\beta$ warmup prevents posterior collapse
    \end{itemize}

    \medskip
    \textcolor{accentorange}{\textbf{Future work:}}
    \begin{itemize}[leftmargin=1em, noitemsep, topsep=1pt]
        \item Adversarial / feature-matching loss for perceptual quality
        \item Larger dataset, more latent dims, longer training time
        %\item Larger latent dim (64+) for diverse datasets
        %\item Streaming / real-time inference
    \end{itemize}

    \medskip
    {\color{subtletxt} \footnotesize \setlength{\baselineskip}{1.1em}
    \textbf{References:} Kingma \& Welling, Auto-Encoding Variational Bayes (2013);
    Caillon \& Esling, RAVE (2021);
    Natsiou et al., Interpretable Timbre Synthesis using VAEs (2023);
    Piczak, ESC-50 Dataset (2015);
    Lostanlen et al., Medley-solos-DB (2019). \par}
